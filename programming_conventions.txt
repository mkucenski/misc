C		= class object (class names/definitions don't require this)
st 	= structure
p 		= pointer
pp 	= pointer to pointer
pos 	= positional reference (e.g. file/stream position pointer or other offset)
str 	= string class
sz		= pointer to first character of string
c	 	= count (e.g. number of bytes read during an operation)
id		= unique identifier (e.g. an inode value)
f		= flag (single bit e.g. boolean)
fx		= flags (multi-bit)
v		= value (number that is otherwise indescribable)
i		= index
rg		= array (range)
dtm	= date/time

m_ 	= class member
